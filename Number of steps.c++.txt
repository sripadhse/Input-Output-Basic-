#include<bits/stdc++.h>


 

using namespace std;


 

int main()

{

    int n;

    cin>>n;

    int ar[n],br[n];


 

    for(int i=0;i<n;i++) cin>>ar[i];

    for(int i=0;i<n;i++) cin>>br[i];


 

    int min=ar[0],coun=0;

    for(int i=0;i<n;i++)

    {

        if(min>ar[i]) min=ar[i]-br[i];   //Find the Minimum by min of Ai - Bi

    }

    

    for(int i=0;i<n;i++)

    {

        while(ar[i]>min && br[i]!=0)

        {

            ar[i]=ar[i]-br[i];     //Then just made every element equals to min by Ai-Bi and count steps

            coun++;

        }

        if(ar[i]!=min) {coun=-1; break; }  //if the Ai is not equals to min then break;

    }

    cout<<coun;

}
